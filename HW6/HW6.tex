function x = usolve(U,y)
    x = zeros(1, length(y));
    n = length(y);
    x(n) = y(n)/U(n, n);
    for i = n-1:-1:1
        tmp = 0;
        for j = i+1:n
            tmp = tmp + U(i, j)*x(j);
        end
        x(i) = (y(i)-tmp) / U(i, i);
    end
    return
end

\end{lstlisting}

function [L, U] = LUNoPivot(A)
    [m,n] = size(A);
    L = eye(size(A));
    U = A;
    for j=1:n-1
       for i=j+1:n
          mult = U(i,j)/U(j,j);
          U(i,:) = U(i,:) - mult*U(j,:);
          L(i, j) = mult;
       end
    end
end

\end{lstlisting}
